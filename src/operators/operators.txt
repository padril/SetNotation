print (arg)
    override null:
        std::cout << '\n';
        return arg;
    override set:
        std::visit([](auto x) {std::cout << x << '\n'; }, arg);
        return arg;
    default:
        std::cout << arg << '\n';
        return arg;

plus (x)
    numeric: return +x;
plus (x, y)
    promote numeric numeric: return x + y;
    string string: return x + y;

minus (x)
    numeric: return -x;
minus (x, y)
    promote numeric numeric: return x - y;

times (x, y)
    promote numeric numeric: return x * y;

slash (x, y)
    override int int: return Rational{x, y};
    promote numeric numeric: return x / y;
