print (arg)
    null:
        std::cout << "NULL\n";
        return arg;
    default:
        std::cout << arg << '\n';
        return arg;

plus (x)
    numeric: return +x;
plus (x, y)
    numeric, numeric: return x + y;
    string, string: return x + y;

times (x, y)
    numeric, numeric: return x * y;

slash (x, y)
    override int, int: return rational{x, y};
    numeric, numeric: return x / y;
